35. Buatlah program untuk menghitung faktorial dengan fungsi. Fungsi faktorial () memberikan return value bertipe long int.. Tolong bantu.! "


Berikut adalah program faktorial dengan cara rekursif dan for-loop:


#include<stdio.h>

long int faktorialRekursif(long int n)

{

if(n<=0)

 return 1;

return n*faktorialRekursif(n-1);

}

long int faktorialLoop(long int n)

{

long int fact = 1;

for(int i=1; i<=n; i++)

{

 fact*=i;

}

return fact;

}

int main()

{

printf("%d\n%d\n",faktorialLoop(10),faktorialRekursif(10));

return 0;

}
